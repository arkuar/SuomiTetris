6   package tetris.game;
7   
8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import javax.swing.Timer;
11  import tetris.domain.Block;
12  import tetris.gui.Refreshable;
13  import tetris.logic.GameBoard;
14  
15  /**
16   * Peliä pyörittävä luokka.
17   *
18   * @author Arttu
19   */
20  public class Tetris extends Timer implements ActionListener {
21  
22      private int width;
23      private int height;
24      private Refreshable refreshable;
25      private GameBoard gameboard;
26      private Block current;
27      public boolean cont;
28      public boolean stopped;
29  
30      /**
31       * Luo pelin annetuilla mitoilla.
32       *
33       * @param height pelilaudan korkeus.
34       * @param width pelilaudan leveys.
35       */
36      public Tetris(int height, int width) {
37          super(400, null);
38          this.width = width;
39          this.height = height;
40          this.gameboard = new GameBoard(height, width);
41          this.current = gameboard.createRandom();
42          this.cont = true;
43          this.stopped = false;
44  
45          super.addActionListener(this);
46      }
47  
48      public int getHeight() {
49          return this.height;
50      }
51  
52      public int getWidth() {
53          return this.width;
54      }
55  
56      public int getX() {
57          return this.gameboard.getX();
58      }
59  
60      public int getY() {
61          return this.gameboard.getY();
62      }
63  
64      public Block getCurrent() {
65          return this.current;
66      }
67  
68      public void setCurrent(Block block) {
69          this.current = block;
70      }
71  
72      public GameBoard getBoard() {
73          return this.gameboard;
74      }
75  
76      public void setRefreshable(Refreshable r) {
77          this.refreshable = r;
78      }
79  
80      /**
81       * Luo uuden pelin.
82       */
83      public void newGame() {
84          this.gameboard.newGame();
85          this.current = gameboard.createRandom();
86          this.cont = true;
87          this.stopped = false;
88          refreshable.refresh();
89          super.start();
90      }
91  
92      /**
93       * Pysäyttää pelin tai jatkaa sitä riippuen tilasta.
94       */
95      public void pause() {
96          if (!stopped) {
97              stopped = true;
98              super.stop();
99          } else {
100             stopped = false;
101             super.start();
102         }
103     }
104 
105     @Override
106     public void actionPerformed(ActionEvent ae) {
107         if (!gameboard.moveDown(current)) {
108             gameboard.blockStopped(current);
109             gameboard.removeLine();
110             setCurrent(gameboard.createRandom());
111             if (!gameboard.canMove(getX(), getY() - 1)) {
112                 super.stop();
113                 this.cont = false;
114             }
115         }
116         refreshable.refresh();
117     }
118 
119 }
