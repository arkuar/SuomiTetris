10  /**
11   * Luokka tarjoaa n채kym채n pelilaudalle.
12   *
13   * @author Arttu
14   */
15  public class GamePanel extends JPanel implements Refreshable {
16  
17      private Tetris game;
18      private Block current;
19      private int blockSize;
20  
21      /**
22       * Luo pelin채kym채n annetulle pelilaudalle.
23       *
24       * @param game Annettu pelilauta.
25       * @param blockSize Palikan koko.
26       */
27      public GamePanel(Tetris game, int blockSize) {
28          super();
29          this.game = game;
30          this.blockSize = blockSize;
31          this.current = null;
32          setBackground(Color.DARK_GRAY);
33      }
34  
35      @Override
36      protected void paintComponent(Graphics g) {
37          super.paintComponent(g);
38          int fromTop = this.getHeight() - game.getHeight() * blockSize;
39          int[][] board = game.getBoard().getBoard();
40  
41          for (int i = 0; i < game.getHeight(); i++) {
42              for (int j = 0; j < game.getWidth(); j++) {
43                  if (board[i][j] == 1) {
44                      paintBlock(g, j * blockSize, fromTop + (game.getHeight() - i - 1) * blockSize, new Color(0, 53, 128));
45                  } else if (board[i][j] == 2) {
46                      paintBlock(g, j * blockSize, fromTop + (game.getHeight() - i - 1) * blockSize, Color.WHITE);
47                  }
48              }
49          }
50  
51          this.current = game.getCurrent();
52          for (int i = 0; i < 4; i++) {
53              int x = game.getX() + current.getX(i);
54              int y = game.getY() - current.getY(i);
55              paintBlock(g, x * blockSize, fromTop + (game.getHeight() - y - 1) * blockSize, game.getBoard().getColor());
56          }
57      }
58  
59      private void paintBlock(Graphics g, int x, int y, Color c) {
60          g.setColor(c);
61          g.fill3DRect(x, y, blockSize, blockSize, true);
62      }
63  
64      @Override
65      public void refresh() {
66          if (!game.cont) {
67              JOptionPane.showMessageDialog(this, "Game Over");
68          }
69          repaint();
70      }
71  
72  }
