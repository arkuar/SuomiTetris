11  /**
12   * Luokka tarjoaa käyttöliittymän pelille.
13   *
14   * @author Arttu
15   */
16  public class UserInterface implements Runnable {
17  
18      private JFrame frame;
19      private GameBoard game;
20      private int height, width;
21      private GamePanel gamepanel;
22  
23      /**
24       * Luo uuden käyttöliittymän parametrina annetulle pelille.
25       *
26       * @param game peli.
27       */
28      public UserInterface(GameBoard game) {
29          this.game = game;
30      }
31  
32      @Override
33      public void run() {
34          frame = new JFrame("SuomiTetris");
35          frame.setPreferredSize(new Dimension(250, 500));
36          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
37          createComponents(frame.getContentPane());
38  
39          frame.pack();
40          frame.setLocationRelativeTo(null);
41          frame.setVisible(true);
42      }
43  
44      private void createComponents(Container container) {
45          this.gamepanel = new GamePanel(game);
46          container.add(gamepanel);
47      }
48  
49  }
