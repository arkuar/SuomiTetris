1   package tetris.gui;
2   
3   import java.awt.BorderLayout;
4   import java.awt.Container;
5   import java.awt.Dimension;
6   import javax.swing.JFrame;
7   import javax.swing.JLabel;
8   import javax.swing.JMenu;
9   import javax.swing.JMenuBar;
10  import javax.swing.JMenuItem;
11  import javax.swing.WindowConstants;
12  import tetris.game.Tetris;
13  
14  /**
15   * Luokka tarjoaa käyttöliittymän pelille.
16   *
17   * @author Arttu
18   */
19  public class UserInterface implements Runnable {
20      
21      private JFrame frame;
22      private Tetris game;
23      private GamePanel gamepanel;
24      private int blockSize;
25      private JLabel scorebar;
26  
27      /**
28       * Luo uuden käyttöliittymän parametrina annetulle pelille.
29       *
30       * @param game peli.
31       * @param blockSize Palikan koko
32       */
33      public UserInterface(Tetris game, int blockSize) {
34          this.game = game;
35          this.blockSize = blockSize;
36      }
37      
38      public JFrame getFrame() {
39          return this.frame;
40      }
41      
42      public JLabel getScorebar() {
43          return this.scorebar;
44      }
45      
46      public Refreshable getRefreshable() {
47          return this.gamepanel;
48      }
49      
50      @Override
51      public void run() {
52          frame = new JFrame("SuomiTetris");
53          int width = (game.getWidth() + 1) * blockSize;
54          int height = (game.getHeight() + 2) * blockSize;
55          frame.setPreferredSize(new Dimension(width, height));
56          frame.setLayout(new BorderLayout());
57          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
58          createComponents(frame.getContentPane());
59          
60          frame.pack();
61          frame.setLocationRelativeTo(null);
62          frame.setVisible(true);
63      }
64      
65      private void createComponents(Container container) {
66          this.gamepanel = new GamePanel(game, blockSize);
67          createMenu(gamepanel);
68          container.add(gamepanel);
69          ButtonListener listener = new ButtonListener(game, game.getCurrent());
70          frame.addKeyListener(listener);
71      }
72      
73      private void createMenu(GamePanel panel) {
74          JMenuBar bar = new JMenuBar();
75          JMenu menugame = new JMenu("Game");
76          JMenuItem newGame = new JMenuItem("New Game");
77          newGame.addActionListener((ae) -> {
78              game.newGame();
79          });
80          menugame.add(newGame);
81          bar.add(menugame);
82          this.scorebar = new JLabel("Score: 0");
83          bar.add(scorebar);
84          panel.setScorebar(scorebar);
85          frame.setJMenuBar(bar);
86      }
87      
88  }
